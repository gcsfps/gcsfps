import time

def contador_regressivo(segundos):
    while segundos:
        mins, secs = divmod(segundos, 60)
        print(f"\r⏳ {mins:02d}:{secs:02d}", end="")
        time.sleep(1)
        segundos -= 1
    print("\n🚀 Tempo encerrado! Hora de agir!")

# Iniciar um contador regressivo de 10 segundos
contador_regressivo(10)
